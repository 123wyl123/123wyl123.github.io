ACWING-3516

Posted on 2021-05-17

题目

[题目链接][1]
给定一个 N×M 的 01 矩阵，矩阵下标从 0 开始。

有 Q 个询问，第 i 个询问为：将矩阵中 (xi,yi) 的元素改成 0 之后，只包含 1 的子矩阵的最大面积是多少。

注意：

1.每次询问均是独立的。
2.询问方格内元素可能本来就是 0。
3.子矩阵的面积是指矩阵的大小。
输入格式
第一行包含两个整数 N,M。

接下来 N 行，每行包含 M 个 01 字符。

再一行包含整数 Q。

接下来 Q 行，每行包含 2 个整数 (xi,yi)。

输出格式
每个询问输出一行一个结果，表示最大面积。

数据范围
对于 20% 的数据，1≤N,M,Q≤10
对于 50% 的数据，1≤N,M,Q≤100
对于 100% 的数据，1≤N,M≤2000,1≤Q≤105, 0≤xi<n,0≤yi<m

输入样例：
4 2
10
11
11
11
3
0 0
2 0
3 1
输出样例：
6
3
4

思路

易知最大矩形必为给出点上下左右任意一个方向上，任取一点，其上方的图像类似于求直方图最大矩形面积，只不过这个问题里面的底边可以为该点上任意的边，所以我们需要枚举底边来求出上方的所有直方图最大矩形，用数组存取该点以上的最大值。类比四个方向，即可以求出答案

代码

​```
import java.util.;
import java.io.;
public class Main{
static int N = 2010;
static int n, m;
static char[][] g;
static int[] l = new int[N], r = new int[N], q = new int[N];
static int[] U = new int[N], D = new int[N], L = new int[N], R = new int[N];
static int[][] s;
public static void main(String[] args) throws Exception{
BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
String[] nm = br.readLine().split(“ “);
n = Integer.parseInt(nm[0]);
m = Integer.parseInt(nm[1]);
g = new char[N][N];

    for(int i = 1; i <= n; ++ i){
        String str = br.readLine();
        char[] temp = str.toCharArray();

        for(int j = 1; j <= m; ++ j) g[i][j] = temp[j - 1];
    }
    init();
    int num = Integer.parseInt(br.readLine());
    while(num -- > 0){
        String[] temp = br.readLine().split(" ");
        //加一个偏移量
        int x = Integer.parseInt(temp[0]) + 1, y = Integer.parseInt(temp[1]) + 1;
        int max = Math.max(Math.max(U[x - 1], D[x + 1]), Math.max(L[y - 1], R[y + 1]));
        System.out.println(max);
    }

}
//预处理4个方向最大矩阵
public static void init(){
    s = new int[N][N];
    for (int i = 1; i <= n; i ++ ){
        for (int j = 1; j <= m; j ++ )
            if (g[i][j] == '1') s[i][j] = s[i - 1][j] + 1;
            else s[i][j] = 0;
        U[i] = Math.max(U[i - 1], calc(s[i], m));
    }
    //情况预处理数组
    s = new int[N][N];
    for (int i = n; i > 0; i -- ){
        for (int j = 1; j <= m; j ++ )
            if (g[i][j] == '1') s[i][j] = s[i + 1][j] + 1;
            else s[i][j] = 0;
        D[i] = Math.max(D[i + 1], calc(s[i], m));
    }
    s = new int[N][N];
    for (int i = 1; i <= m; i ++ ){
        for (int j = 1; j <= n; j ++ )
            if (g[j][i] == '1') s[i][j] = s[i - 1][j] + 1;
            else s[i][j] = 0;
        L[i] = Math.max(L[i - 1], calc(s[i], n));
    }

    s = new int[N][N];
    for (int i = m; i > 0; i -- ){
        for (int j = 1; j <= n; j ++ )
            if (g[j][i] == '1') s[i][j] = s[i + 1][j] + 1;
            else s[i][j] = 0;
        R[i] = Math.max(R[i + 1], calc(s[i], n));
    }
}

//计算矩形最大面积
private static int calc(int[] h, int n){
    h[0] = h[n + 1] = -1;

    int tt = 0;
    q[0] = 0;
    for (int i = 1; i <= n; i ++ )
    {
        while (h[q[tt]] >= h[i]) tt -- ;
        l[i] = q[tt];
        q[ ++ tt] = i;
    }

    tt = 0;
    q[0] = n + 1;
    for (int i = n; i > 0; i -- )
    {
        while (h[q[tt]] >= h[i]) tt -- ;
        r[i] = q[tt];
        q[ ++ tt] = i;
    }

    int res = 0;
    for (int i = 1; i <= n; i ++ )
        res = Math.max(res, h[i] * (r[i] - l[i] - 1));
    return res;
}

}
